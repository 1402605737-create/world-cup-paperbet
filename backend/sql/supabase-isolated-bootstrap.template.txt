do $bootstrap$
begin
  if exists (select 1 from pg_roles where rolname = 'world_cup_paperbet_app') then
    raise exception '角色 world_cup_paperbet_app 已存在；为避免修改已有角色，操作已停止';
  end if;

  if exists (select 1 from information_schema.schemata where schema_name = 'world_cup_paperbet') then
    raise exception 'Schema world_cup_paperbet 已存在；为避免修改已有 Schema，操作已停止';
  end if;

  create role world_cup_paperbet_app
    login
    password '__APP_PASSWORD__'
    nosuperuser
    nocreatedb
    nocreaterole
    noinherit
    nobypassrls;
end
$bootstrap$;

create schema world_cup_paperbet;

create table world_cup_paperbet.demo_cases (
  id uuid primary key default gen_random_uuid(),
  slug text unique not null,
  title text not null,
  status text not null check (status in ('ready', 'blocked', 'verified')),
  agent_trace jsonb not null default '[]'::jsonb,
  evidence jsonb not null default '[]'::jsonb,
  result jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table world_cup_paperbet.ai_call_logs (
  id uuid primary key default gen_random_uuid(),
  provider text not null,
  model text not null,
  request_id text,
  fallback boolean not null default false,
  response_json jsonb not null,
  created_at timestamptz not null default now()
);

create index ai_call_logs_created_at_idx
  on world_cup_paperbet.ai_call_logs (created_at desc);

grant connect on database postgres to world_cup_paperbet_app;
grant usage on schema world_cup_paperbet to world_cup_paperbet_app;
grant select on world_cup_paperbet.demo_cases to world_cup_paperbet_app;
grant select, insert on world_cup_paperbet.ai_call_logs to world_cup_paperbet_app;

create policy world_cup_paperbet_demo_cases_read
  on world_cup_paperbet.demo_cases
  for select
  to world_cup_paperbet_app
  using (true);

create policy world_cup_paperbet_ai_call_logs_read
  on world_cup_paperbet.ai_call_logs
  for select
  to world_cup_paperbet_app
  using (true);

create policy world_cup_paperbet_ai_call_logs_insert
  on world_cup_paperbet.ai_call_logs
  for insert
  to world_cup_paperbet_app
  with check (provider = 'deepseek' and fallback = false);
