-- Run once as a Supabase administrator in SQL Editor.
-- Replace __APP_PASSWORD__ only in a local ignored temporary copy.
-- This script intentionally does not inspect or modify any existing application object.

do $bootstrap$
begin
  if exists (select 1 from pg_roles where rolname = 'world_cup_paperbet_app') then
    raise exception 'Role world_cup_paperbet_app already exists; stop to avoid modifying an existing role';
  end if;

  if exists (select 1 from information_schema.schemata where schema_name = 'world_cup_paperbet') then
    raise exception 'Schema world_cup_paperbet already exists; stop to avoid modifying an existing schema';
  end if;

  create role world_cup_paperbet_app
    login
    password '__APP_PASSWORD__'
    nosuperuser
    nocreatedb
    nocreaterole
    noinherit
    nobypassrls;
end
$bootstrap$;

create schema world_cup_paperbet;

create table world_cup_paperbet.demo_cases (
  id uuid primary key default gen_random_uuid(),
  slug text unique not null,
  title text not null,
  status text not null check (status in ('ready', 'blocked', 'verified')),
  agent_trace jsonb not null default '[]'::jsonb,
  evidence jsonb not null default '[]'::jsonb,
  result jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table world_cup_paperbet.ai_call_logs (
  id uuid primary key default gen_random_uuid(),
  provider text not null,
  model text not null,
  request_id text,
  fallback boolean not null default false,
  response_json jsonb not null,
  created_at timestamptz not null default now()
);

create index ai_call_logs_created_at_idx
  on world_cup_paperbet.ai_call_logs (created_at desc);

insert into world_cup_paperbet.demo_cases (slug, title, status, agent_trace, evidence, result)
values (
  'configuration-readiness',
  'Data source configuration readiness',
  'ready',
  '[
    {"step": 1, "agent": "Config Inspector", "action": "Inspect server-side environment flags"},
    {"step": 2, "agent": "Compliance Guard", "action": "Block unconfigured providers from returning invented data"},
    {"step": 3, "agent": "DeepSeek Verifier", "action": "Run a real structured-output connectivity check"}
  ]'::jsonb,
  '[
    "Sports and odds providers expose explicit configured flags",
    "The backend never returns mock match, result, or odds data",
    "AI verification records fallback=false responses"
  ]'::jsonb,
  '{"outcome": "Configuration dashboard is ready for live-provider onboarding"}'::jsonb
);

grant connect on database postgres to world_cup_paperbet_app;
grant usage on schema world_cup_paperbet to world_cup_paperbet_app;
grant select on world_cup_paperbet.demo_cases to world_cup_paperbet_app;
grant select, insert on world_cup_paperbet.ai_call_logs to world_cup_paperbet_app;

alter table world_cup_paperbet.demo_cases enable row level security;
alter table world_cup_paperbet.ai_call_logs enable row level security;

create policy world_cup_paperbet_demo_cases_read
  on world_cup_paperbet.demo_cases
  for select
  to world_cup_paperbet_app
  using (true);

create policy world_cup_paperbet_ai_call_logs_read
  on world_cup_paperbet.ai_call_logs
  for select
  to world_cup_paperbet_app
  using (true);

create policy world_cup_paperbet_ai_call_logs_insert
  on world_cup_paperbet.ai_call_logs
  for insert
  to world_cup_paperbet_app
  with check (provider = 'deepseek' and fallback = false);

select count(*) as demo_case_count from world_cup_paperbet.demo_cases;
